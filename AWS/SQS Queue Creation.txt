export AWS_PROFILE_NAME=cna-lab-1
aws sts get-caller-identity --profile $AWS_PROFILE_NAME
export AWS_REGION=us-east-1
export AWS_ACCESS_KEY_ID="$(aws --profile "$AWS_PROFILE_NAME" configure get aws_access_key_id)"
echo $AWS_ACCESS_KEY_ID
export AWS_SECRET_ACCESS_KEY="$(aws --profile "$AWS_PROFILE_NAME" configure get aws_secret_access_key)"
echo $AWS_SECRET_ACCESS_KEY

export AWS_SQS_QUEUE_NAME=mkpublisher-queue
aws --profile "$AWS_PROFILE_NAME" --region "$AWS_REGION" sqs create-queue --queue-name "$AWS_SQS_QUEUE_NAME"
export AWS_SQS_QUEUE_URL="$(aws --profile "$AWS_PROFILE_NAME" --region "$AWS_REGION" sqs get-queue-url --queue-name "$AWS_SQS_QUEUE_NAME" --output text)"
echo $AWS_SQS_QUEUE_URL
